Sequência de escape ANSI

echo "\033[2J\033[;H";

  • \033: Indica que uma sequência de escape está começando. O código \033 é usado para representar o caractere “ESC” (Escape), que tem um valor ASCII decimal de 27 (ou octal 033).
  • [2J: Essa parte da sequência de escape é usada para limpar a tela. O 2J é uma instrução que significa “limpar toda a tela” (J de “Janela”). O número entre colchetes (2) indica o tipo de limpeza, neste caso, “limpar a tela inteira”.
  • \033: Indicando novamente que uma outra sequência de escape está começando.
  • [;H: Esta parte é usada para posicionar o cursor na tela. Neste caso, a instrução ;H é usada para definir a posição do cursor como a posição padrão do início da tela.

<?php 

abstract class Matriz {
    protected $linhas;
    protected $colunas;
    protected $matriz;

    public function __construct(int $linhas, int $colunas, $valorPadrao) {
        if ($linhas < 0 || $colunas < 0) {
            throw new InvalidArgumentException("O número de linhas e colunas deve ser maior ou igual a zero.");
        }

        $this->linhas = $linhas;
        $this->colunas = $colunas;
        $this->matriz = array_fill(0, $linhas, array_fill(0, $colunas, $valorPadrao));
    }

    // Método abstrato para validação de tipo, que será implementado pelas classes filhas
    abstract protected function validarTipo($valor): bool;

    // Acesso ao valor de um índice
    public function get($linha, $coluna) {
        if (!$this->indiceValido($linha, $coluna)) {
            throw new OutOfBoundsException("Erro ao acessar valor: índice inválido.");
        }
        return $this->matriz[$linha][$coluna];
    }

    // Definir o valor de um índice
    public function set($linha, $coluna, $valor) {
        if (!$this->indiceValido($linha, $coluna)) {
            throw new OutOfBoundsException("Erro ao definir valor: índice inválido.");
        }

        if (!$this->validarTipo($valor)) {
            throw new InvalidArgumentException("Erro ao definir valor: tipo inválido.");
        }

        $this->matriz[$linha][$coluna] = $valor;
    }

    private function indiceValido($linha, $coluna): bool {
        return $linha >= 0 && $linha < $this->linhas && $coluna >= 0 && $coluna < $this->colunas;
    }

    // Visualização da matriz
    public function visualizarMatriz() {
        for ($i = 0; $i < $this->linhas; $i++) {
            for ($j = 0; $j < $this->colunas; $j++) {
                echo $this->matriz[$i][$j] . " ";
            }
            echo "\n";
        }
    }
}

class MatrizInt extends Matriz {
    public function __construct(int $linhas, int $colunas, $valorPadrao = 0) {
        parent::__construct($linhas, $colunas, $valorPadrao);
    }

    protected function validarTipo($valor): bool {
        return is_int($valor);
    }
}

class MatrizFloat extends Matriz {
    public function __construct(int $linhas, int $colunas, $valorPadrao = 0.0) {
        parent::__construct($linhas, $colunas, $valorPadrao);
    }

    protected function validarTipo($valor): bool {
        return is_float($valor);
    }
}

class MatrizString extends Matriz {
    public function __construct(int $linhas, int $colunas, $valorPadrao = "") {
        parent::__construct($linhas, $colunas, $valorPadrao);
    }

    protected function validarTipo($valor): bool {
        return is_string($valor);
    }
}
function main() {
    // Criar uma matriz de inteiros
    $matrizInt = new MatrizInt(3, 3, 0);  // Inicializa com o valor 0

    // Definindo valores na matriz de inteiros
    try {
        echo "Setando valor 5 na posição (1, 1) da Matriz de Inteiros.\n";
        $matrizInt->set(1, 1, 5);  // Definindo um valor válido para a posição (1, 1)
        echo "Valor definido com sucesso.\n";
    } catch (Exception $e) {
        echo $e->getMessage() . "\n";
    }

    // Visualizando a matriz de inteiros
    echo "Visualizando Matriz de Inteiros:\n";
    $matrizInt->visualizarMatriz();

    // Criar uma matriz de floats
    $matrizFloat = new MatrizFloat(2, 2, 0.0);  // Inicializa com o valor 0.0

    // Definindo valores na matriz de floats
    try {
        echo "Setando valor 1.5 na posição (0, 1) da Matriz de Floats.\n";
        $matrizFloat->set(0, 1, 1.5);  // Definindo um valor válido para (0, 1)
        echo "Valor definido com sucesso.\n";
    } catch (Exception $e) {
        echo $e->getMessage() . "\n";
    }

    // Visualizando a matriz de floats
    echo "Visualizando Matriz de Floats:\n";
    $matrizFloat->visualizarMatriz();

    // Criar uma matriz de strings
    $matrizString = new MatrizString(2, 2, "");  // Inicializa com valor vazio

    // Definindo valores na matriz de strings
    try {
        echo "Setando valor 'Hello' na posição (1, 0) da Matriz de Strings.\n";
        $matrizString->set(1, 0, "Hello");  // Definindo um valor válido para (1, 0)
        echo "Valor definido com sucesso.\n";
    } catch (Exception $e) {
        echo $e->getMessage() . "\n";
    }

    // Visualizando a matriz de strings
    echo "Visualizando Matriz de Strings:\n";
    $matrizString->visualizarMatriz();
}
